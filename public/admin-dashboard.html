SQL Setup Reference

Run these in your Supabase SQL Editor if tables are missing:

-- See SQL box on right →
Required SQL Schema
-- template_configs table
CREATE TABLE IF NOT EXISTS public.template_configs (
  id TEXT PRIMARY KEY,
  name TEXT NOT NULL,
  price NUMERIC NOT NULL DEFAULT 499,
  placeholder_url TEXT,
  is_active BOOLEAN DEFAULT TRUE,
  is_premium BOOLEAN DEFAULT FALSE,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- customer_invitations table
CREATE TABLE IF NOT EXISTS public.customer_invitations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id),
  user_email TEXT,
  template_type TEXT,
  title TEXT,
  status TEXT DEFAULT 'draft',
  payment_id TEXT,
  order_id TEXT,
  rsvps JSONB DEFAULT '[]',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- guest_list table
CREATE TABLE IF NOT EXISTS public.guest_list (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id),
  name TEXT NOT NULL,
  phone TEXT NOT NULL,
  status TEXT DEFAULT 'Added',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Enable RLS
ALTER TABLE template_configs ENABLE ROW LEVEL SECURITY;
ALTER TABLE customer_invitations ENABLE ROW LEVEL SECURITY;
ALTER TABLE guest_list ENABLE ROW LEVEL SECURITY;

-- Admin can read all (set role=admin in user_metadata)
CREATE POLICY "Admin all" ON customer_invitations
  FOR ALL USING (true);
CREATE POLICY "Admin all" ON guest_list
  FOR ALL USING (true);
CREATE POLICY "Public read templates" ON template_configs
  FOR SELECT USING (true);
CREATE POLICY "Admin write templates" ON template_configs
  FOR ALL USING (true);
