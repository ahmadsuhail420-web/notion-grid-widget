      function showToast(message, type = 'success') {
          const container = document.getElementById('toast-container');
          const toast = document.createElement('div');
          
          const bgColor = type === 'success' ? 'bg-[#9c404d]' : 'bg-red-600';
          
          toast.className = `toast px-6 py-4 rounded-xl text-white font-bold shadow-2xl ${bgColor} flex items-center space-x-3`;
          toast.innerHTML = `
              <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                  ${type === 'success' ? 
                      '<path stroke-linecap="round" stroke-linejoin="round" stroke-width="3" d="M5 13l4 4L19 7"></path>' : 
                      '<path stroke-linecap="round" stroke-linejoin="round" stroke-width="3" d="M6 18L18 6M6 6l12 12"></path>'}
              </svg>
              <span>${message}</span>
          `;
          
          container.appendChild(toast);
          setTimeout(() => toast.classList.add('show'), 10);
          setTimeout(() => {
              toast.classList.remove('show');
              setTimeout(() => toast.remove(), 300);
          }, 3000);
      }

      function handleCardClick(card) {
        if (window.innerWidth >= 1024) return;
        
        const isActive = card.classList.contains('mobile-active');
        
        // Close all other cards
        document.querySelectorAll('.template-card').forEach(c => {
          c.classList.remove('mobile-active');
        });
        
        if (!isActive) {
          card.classList.add('mobile-active');
        }
      }
      
      // Close cards when clicking outside
      document.addEventListener('click', (e) => {
        if (!e.target.closest('.template-card')) {
          document.querySelectorAll('.template-card').forEach(c => {
            c.classList.remove('mobile-active');
          });
        }
      });

      const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
          if (entry.isIntersecting) {
            entry.target.classList.add('active');
          } else {
            entry.target.classList.remove('active');
          }
        });
      }, { threshold: 0.1 });

      document.querySelectorAll('.scroll-reveal, .scroll-pop').forEach(el => observer.observe(el));

    <script>
      let sb = null;
      async function initSupabase() {
        try {
          const res = await fetch('/api/config');
          if (!res.ok) return;
          const config = await res.json();
          if (config.supabaseUrl && config.supabaseAnonKey) {
            sb = supabase.createClient(config.supabaseUrl, config.supabaseAnonKey);
            const { data: { session } } = await sb.auth.getSession();
            if (session) {
              document.getElementById('nav-login-link').classList.add('hidden');
              document.getElementById('nav-dashboard-link').classList.remove('hidden');
            }
            loadTemplateCardData();
          }
        } catch(e) { console.error('Supabase init failed', e); }
      }
      initSupabase();

      async function loadTemplateCardData() {
        if (!sb) return;
        try {
          const { data, error } = await sb.from('template_configs').select('id, name, price, original_price, is_premium, placeholder_url');
          if (error || !data) return;
          data.forEach(t => {
            // Update name display
            const nameEl = document.getElementById('name-' + t.id);
            if (nameEl && t.name) nameEl.textContent = t.name;

            // Handle preview or image
            const previewContainer = document.getElementById('preview-' + t.id);
            if (previewContainer && t.placeholder_url) {
              previewContainer.innerHTML = `<img src="${t.placeholder_url}" class="w-full h-full object-cover">`;
            }

            // Update price display
            const priceEl = document.getElementById('price-' + t.id);
            if (priceEl) priceEl.textContent = '₹' + t.price;

            // Calculate and display offer %
            const offerEl = document.getElementById('offer-' + t.id);
            const origPriceEl = document.getElementById('original-price-' + t.id);
            
            if (offerEl && t.price) {
              const originalPrice = t.original_price || 1499;
              if (origPriceEl) {
                origPriceEl.textContent = '₹' + originalPrice;
                origPriceEl.style.display = 'inline-block';
              }
              
              if (t.price < originalPrice) {
                const percent = Math.round(((originalPrice - t.price) / originalPrice) * 100);
                offerEl.textContent = percent + '% OFF';
                offerEl.style.display = 'inline-block';
              } else {
                offerEl.style.display = 'none';
                if (origPriceEl) origPriceEl.style.display = 'none';
              }
            }

            // Update badge: golden crown if is_premium
            const badgeEl = document.getElementById('badge-' + t.id);
            if (badgeEl) {
              if (t.is_premium) {
                badgeEl.innerHTML = `<span class="flex items-center gap-1 bg-gradient-to-r from-[#d4af37] to-[#b8962e] text-white text-[10px] px-2.5 py-1 rounded-full font-bold shadow-lg">
                  <svg xmlns="http://www.w3.org/2000/svg" class="w-3 h-3" viewBox="0 0 24 24" fill="currentColor"><path d="M2 19l2-7 5 4 3-9 3 9 5-4 2 7H2z"/></svg>
                  Premium
                </span>`;
              } else {
                badgeEl.innerHTML = '';
              }
            }
          });
        } catch(e) { console.error('loadTemplateCardData error', e); }
      }

      async function handleSelectTemplate(templateId) {
        if (!sb) {
          window.location.href = '/login?redirect=' + encodeURIComponent('/payment?template=' + templateId);
          return;
        }
        const { data: { session } } = await sb.auth.getSession();
        if (!session) {
          window.location.href = '/login?redirect=' + encodeURIComponent('/payment?template=' + templateId);
        } else {
          // Check if admin to provide better UX
          const admins = ['ahmadsuhail420@gmail.com', 'suhailcloud7@gmail.com'];
          if (admins.includes(session.user.email)) {
             // Admin still goes to payment page but it will be 0, or we can go directly
             // Let's go to payment so they can confirm details
             window.location.href = '/payment?template=' + templateId;
          } else {
             window.location.href = '/payment?template=' + templateId;
          }
        }
      }
